# 使用说明

使用 import 在页面引用该组件。

<template>
    <el-button @click="returnFun">点我回显</el-button> <!--触发回显数据的方法-->
    <region-cascader ref="regionArr" v-model="regionArr" @regionObj="event" />
</template>
<script>
    import regionCascader from "@/components/Cascader/regionCascader";
    export default {
        components: {
            regionCascader
        },
        data() {
            return {
                regionArr: [1, 3, 18]
            };
        },
        methods: {
            // 回显数据
            returnFun() {
                this.$refs.regionArr.returnRegion(this.regionArr); // 指向组件标签上的ref名称并触发组件里的方法
            },
            // 选中行业触发事件，带入选中的值
            event(nameArr, idArr) {
                console.log(nameArr, idArr);
            }
        }
    };
</script>
