<div class="info-con" style="border: none">

        :is-img="true"
        type="text"
        txt="上传照片"
        btn-icon="el-icon-plus"
        :limit="6"
        :single-limit="0"

        <file-list :fileList="commImgList" class="file-style" list-type="picture" mode="edit" @deleteFile="deletePhoto" />
    </div>
</div>

        return {
            imgList: [],
            commImgList: [],
        }
    }
}

</script>
