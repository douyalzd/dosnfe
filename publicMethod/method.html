                  # 全局公用方法

                  # 全局下载附件

                  $downLoadFile(fileUrl)

                  参数： {String} fileUrl 下载路径

                  <!-- 标签使用方法 -->
<span 
    class="accessory"
    v-for="(item, index) in fileList"
    :key="index"
    @click="$downLoadFile(item.eurl, item.ename)">{{item.ename}}</span>

                  // js用法
...
methods: {
    test() {
        this.$downLoadFile(eurl, ename);
    }
}
...

                  $downLoadByUrl(fileUrl, param)

                  参数：

                  • {String} fileUrl 接口路径
                  • {Object} param 接口所需数据
                  // js用法
...
methods: {
    test() {
        this.$downLoadByUrl(apiurl, data);
    }
}
...

                  注意

                  • {Object} paramsStr 接口所需数据
                  // js用法
...
methods: {
    test() {
        this.$exportFile(apiurl, data);
    }
}

                  # 时间选择框日期限制选择

                  $dateCompare(isStart, nowTime, isEqual = false)

                  参数：

                  • {String} isStart 是开始时间还是结束时间 'start'；'end'
                  • {Date} nowTime 输入框绑定的时间 如果是【开始时间】 传 【结束时间v-model】，如果是【结束时间】 传 【开始时间
                  • {Boolean} isEqual 开始时间和结束时间是否可相等 默认是false
                  • 使用示例 在时间组件上加属性
                  <el-date-picker 
    :editable="false"
    v-model="fliterForm.beginTime"
    :picker-options="$dateCompare('start', fliterForm.endTime, false)"
    type="date" placeholder="开始时间"
    value-format="yyyy-MM-dd" />
<span>-</span>
<el-date-picker 
    :editable="false"
    v-model="fliterForm.endTime"
    :picker-options="$dateCompare('end', fliterForm.beginTime, false)"
    type="date" placeholder="结束时间"
    value-format="yyyy-MM-dd" />

                  # 选择时间范围限制

                  $dateRangeSel(beginTime, endTime, isEqual = false)
                  控制只能选择某个范围内的时间

                  参数：

                  • {String} beginTime 开始时间
                  • {String} endTime 结束时间
                  • {Boolean} isEqual 否可选与beginTime和endTime相同的时间
                  <el-date-picker 
    :editable="false"
    v-model="fliterForm.beginTime"
    :picker-options="$dateRangeSel(fliterForm.beginTime, fliterForm.endTime, false)"
    type="date" placeholder="开始时间"
    value-format="yyyy-MM-dd" />

                  # 公用中文排序方法

                  $sortByZh(key, a, b)

                  注意

                  只适用于el-table组件

                  参数：

                  • {String} key 需要排序的字段
                  • {Object} a 组件自带的
                  • {Object} b 组件自带的
                  <el-table-column prop="communityName" label="社区名称" sortable :sort-method="$sortByZh.bind(null,'communityName')" />
